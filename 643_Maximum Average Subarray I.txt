class Solution {
public:
    double findMaxAverage(vector<int>& nums, int k) {
        double  sum=0, ret;

        for (int i=0;i<nums.size();i++) {
            if ( i< k) {
                sum+=nums[i]; // replace with nums.at(i) if want more safe
                ret = sum;
            }
            else {
                sum += nums[i] - nums[i-k]; 
                if (sum > ret) ret = sum;
            }
        }
        return ret/k;
    }
};
